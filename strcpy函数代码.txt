#include<stdio.h>
#include<string.h>
#include<assert.h>
char* strcpy(char* dest, const char* src)
{
	//指针断言
	assert(dest != NULL);
	assert(src!= NULL);
	//由于在赋值过程中，dest的指向会被更改
	//所以先保留首元素地址
	char* ret = dest;
	while (*dest++ = *src++)
	{
		;
	}
	return ret;
}
